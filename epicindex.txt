  function renderPoem(index) {
    const poem = poems[index];
    const likeKey = `likes_${index}`;
    const commentKey = `comments_${index}`;
    const likes = parseInt(localStorage.getItem(likeKey)) || 0;
    const comments = JSON.parse(localStorage.getItem(commentKey)) || [];

    container.innerHTML = `
      <div class="poem-full active">
        <h2>${poem.title}</h2>
        <img src="${poem.image}" alt="${poem.title}">
        <div class="poem-text">${poem.text}</div>

        <div class="like-comment">
          <button class="like-button" onclick="likePoem(${index})">❤️ Like (<span id="likeCount${index}">${likes}</span>)</button>

          <div class="comments">
            <textarea id="commentInput${index}" placeholder="Write a comment..."></textarea>
            <button onclick="addComment(${index})">Post Comment</button>
            <div class="comment-list" id="commentList${index}">
              ${comments.map(c => `<div class="comment">${c}</div>`).join('')}
            </div>
          </div>
        </div>

        <div class="button-row">
          <button class="btn" onclick="goBack()">← Back to List</button>
          <button class="btn" onclick="showPoem(${(index + 1) % poems.length})">Next →</button>
        </div>
      </div>
    `;
  }

  function likePoem(index) {
    const key = `likes_${index}`;
    let count = parseInt(localStorage.getItem(key)) || 0;
    count++;
    localStorage.setItem(key, count);
    document.getElementById(`likeCount${index}`).innerText = count;
  }

  function addComment(index) {
    const input = document.getElementById(`commentInput${index}`);
    const list = document.getElementById(`commentList${index}`);
    const key = `comments_${index}`;
    const comments = JSON.parse(localStorage.getItem(key)) || [];

    if (input.value.trim() !== "") {
      comments.push(input.value.trim());
      localStorage.setItem(key, JSON.stringify(comments));
      list.innerHTML += `<div class="comment">${input.value.trim()}</div>`;
      input.value = "";
    }
  }
